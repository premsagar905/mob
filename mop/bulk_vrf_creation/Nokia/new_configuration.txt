            policy-statement "{{vrf_name}}_Import"
                @entry {{import_entry_number}}
                    @from
                        @protocol bgp-vpn
                       @community "target:{{import_rt}}"
                    @exit
                    @action accept
                    exit
             exit all

           policy-statement "{{vrf_name}}_Export"
                entry {{import_entry_number}}
                    from
                        protocol direct
                    exit
                    to
                        protocol bgp-vpn
                    exit
                    action accept
                         %community add {{export_rt}}
                    exit
                exit all
